### 接收post请求数据

```javascript
const http = require('http');

http.createServer((req,res)=>{
    console.log(req.method);
    let postData = '';
    // post请求得做事件监听
    req.on('data',(data)=>{
        postData += data;
        console.log(postData);
    })
    // 监听数据接收完毕
    req.on('end',()=>{

    })
}).listen(3000,'127.0.0.1');
```

queryString可以直接更好的获取请求值
